% E1 â€” hoja de ruta de la secciĂłn (OJO: se solapa con el Sumario de Felipe)
\begin{frame}{La secciĂłn: quĂ© se deriva}
  Se construye la \resultado{regla de decisiĂłn Ăłptima} y el valor de la
  oportunidad de inversiĂłn, en tres pasos:
  \par\vspace{3mm}
  \begin{enumerate}
    \item \rj{$V$} y \az{$F$} siguen movimientos brownianos geomĂ©tricos.
    \item Se obtiene la \concepto{tasa de descuento correcta} cuando los
          inversores son reacios al riesgo.
    \item Se admite que \rj{$V_t$} pueda caer repentinamente a cero.
  \end{enumerate}
\end{frame}

% E2 â€” caso F constante, umbral Ct*
\begin{frame}{Umbral de inversiĂłn: el caso $F$ constante}
  Para introducir el problema, \rj{$V_t$} sigue un MBG y \az{$F$} es
  constante. La firma recibe \rj{$V_t$}$-$\az{$F$} al invertir, y la regla
  Ăłptima es un \concepto{umbral}: invertir cuando
  \rj{$V_t$}$/$\az{$F$} $\geq$ \rj{$C_t^*$}.
  \par\vspace{2.5mm}
  Si la oportunidad vence en $T$: allĂ­ conviene invertir si
  \rj{$V_T$}$\geq$\az{$F$}, o sea \rj{$C_T^*$}$=1$; hacia atrĂˇs en el tiempo,
  cada instante $t$ tiene su umbral \rj{$C_t^*$}.
  \par\vspace{2.5mm}
  En \rj{$V_t$}$/$\az{$F$}$=$\rj{$C_t^*$}, el VAN es
  \az{$F$}$($\rj{$C_t^*$}$-1)$. Para una barrera arbitraria, el valor de la
  oportunidad es el valor presente esperado del payoff:
  \begin{ecmed}
    \rj{X(T)} = \az{E_0\bigl\{ e^{-\mu t'}\,( V_{t'} - F ) \bigr\}}
  \end{ecmed}
\end{frame}

% E3 â€” vida infinita
\begin{frame}{Vida infinita: el umbral se vuelve constante}
  Cuando la oportunidad tiene \concepto{vida infinita} ($T\to\infty$)
  desaparece el tiempo calendario: el umbral no puede depender de $t$, asĂ­
  que \rj{$C_t^*$}$=$\rj{$C^*$} para todo $t$.
  \par\vspace{4mm}
  Entonces \rj{$V_{t'}$}$-$\az{$F$}$=$\az{$F$}$($\rj{$C^*$}$-1)$ es constante,
  y maximizar \rj{$X(T)$} se reduce a
  \begin{ec}
    \max_{C'}\; \az{F\,(C' - 1)\; E_0\bigl\{ e^{-\mu t'} \bigr\}}
  \end{ec}
  Es un caso especial del problema general que sigue.
\end{frame}

% E4 â€” modelo general + homogeneidad
\begin{frame}{Modelo general: $F$ tambiĂ©n estocĂˇstico}
  La caracterizaciĂłn del umbral ya no es evidente, pero la regla sigue siendo
  de umbral sobre el ratio \rj{$V$}$/$\az{$F$}.
  \par\vspace{3mm}
  \concepto{Homogeneidad de grado cero}: con \rj{$V'=kV$} y \az{$F'=kF$},
  \begin{ecmed}
    \rj{\frac{dV'}{V'}} = \az{\alpha_v\,dt + \sigma_v\,dz_v}
    \qquad
    \az{\frac{dF'}{F'} = \alpha_f\,dt + \sigma_f\,dz_f}
  \end{ecmed}
  el problema es formalmente idĂ©ntico, de modo que el umbral no depende de $k$
  (es homogĂ©neo de grado cero en \rj{$V$}, \az{$F$}) ni del calendario cuando
  \mbox{$T\to\infty$}.
  \begin{cajatexto}
    Regla Ăłptima: invertir cuando \rj{$V_t$}$/$\az{$F_t$} $\geq$ \rj{$C^*$}.
  \end{cajatexto}
\end{frame}

% E5 â€” la soluciĂłn (X, epsilon)
\begin{frame}{La soluciĂłn del modelo general}
  El apĂ©ndice demuestra que la regiĂłn de inversiĂłn es
  $[$\rj{$C^*$}$,\infty)$. Como el factor \rj{$C^*\!-1$} es constante, sale de
  la esperanza:
  \begin{ecmed}
    \az{E_0\bigl\{ F_{t'}\,}\rj{[C^*\!-1]}\az{\,e^{-\mu t'} \bigr\}}
      = \rj{[C^*\!-1]}\,\az{E_0\bigl\{ F_{t'}\,e^{-\mu t'} \bigr\}}
  \end{ecmed}
  y el valor de la oportunidad de inversiĂłn resulta
  \begin{ec}
    \rj{X} = \az{(C^* - 1)\,F_0
      \left(\frac{V_0/F_0}{C^*}\right)^{\!\varepsilon}}
  \end{ec}
  El exponente \rj{$\varepsilon$} es la raĂ­z positiva de la ecuaciĂłn
  caracterĂ­stica del apĂ©ndice.
\end{frame}

% E5b â€” el exponente epsilon (separado de la soluciĂłn para que entre)
\begin{frame}{El exponente $\varepsilon$}
  Depende de las derivas \az{$\alpha_v,\alpha_f$}, la varianza conjunta
  \az{$\sigma^2$} y la tasa de descuento \az{$\mu$}:
  \par\vspace{2mm}
  \begin{ec}
    \rj{\varepsilon} = \az{\sqrt{\left(\frac{\alpha_v-\alpha_f}{\sigma^2}
      - \frac12\right)^{\!2} + \frac{2(\mu-\alpha_f)}{\sigma^2}}
      + \left(\frac12 - \frac{\alpha_v-\alpha_f}{\sigma^2}\right)}
  \end{ec}
  con \az{$\sigma^2 = \sigma_v^2 + \sigma_f^2 - 2\rho_{vf}\sigma_v\sigma_f$} la
  varianza del ratio \rj{$V$}$/$\az{$F$}.
  \par\vspace{4mm}
  La condiciĂłn \az{$\alpha_v<\mu$} garantiza \rj{$\varepsilon>1$}, y por lo
  tanto \rj{$C^*>1$}.
\end{frame}

% E6 â€” apĂ©ndice: la EDP de Feynman-Kac + condiciones de contorno
\begin{frame}{Desarrollo matemĂˇtico (apĂ©ndice)}
  El valor \az{$L(V,F)=E_0\{F_{t'}\,e^{-\mu t'}\}$}, con \rj{$t'$} el primer
  instante en que \rj{$V$}$/$\az{$F$} toca \rj{$C^*$}, satisface por
  \concepto{Feynman--Kac} la EDP (A.2):
  \begin{ecmed}
    \az{\mu L = \tfrac12\!\left[L_{vv}V^2\sigma_v^2 + L_{ff}F^2\sigma_f^2
      + 2 L_{vf}VF\sigma_{vf}\right] + L_v \alpha_v V + L_f \alpha_f F}
  \end{ecmed}
  Condiciones de contorno (se decide invertir en $[$\rj{$C^*$}$,\infty)$):
  \begin{itemize}
    \item[(i)] \az{$L=F$} en la barrera \rj{$V$}$/$\az{$F$}$=$\rj{$C^*$}: aquĂ­
          \rj{$t'=0$} y se invierte.
    \item[(ii)] \az{$L\to0$} cuando \rj{$V$}$/$\az{$F$}$\to0$: \rj{$t'\to\infty$},
          nunca se invierte.
  \end{itemize}
\end{frame}

% E6b â€” apĂ©ndice: forma funcional â†’ ecuaciĂłn caracterĂ­stica
\begin{frame}{ApĂ©ndice: la ecuaciĂłn caracterĂ­stica}
  Se propone la forma funcional \az{$L = k\,F^{a}\,C^{b}$}, con
  \az{$C=$}\rj{$V$}$/$\az{$F$}. La condiciĂłn (i) fija \az{$a=1$} y
  \az{$k=(C^*)^{-b}$}; sustituyendo en (A.2) se obtiene la ecuaciĂłn
  caracterĂ­stica (A.4):
  \begin{ec}
    \az{\mu = \tfrac12\, b\,(b-1)\,\sigma^2 + b\,\alpha_v + (1-b)\,\alpha_f}
  \end{ec}
  La condiciĂłn (ii) exige \az{$b>0$}: la raĂ­z positiva es \rj{$b=\varepsilon$},
  el exponente que aparece en \rj{$X$}.
\end{frame}

% E7 â€” C* Ăłptimo
\begin{frame}{El umbral Ăłptimo $C^* = \varepsilon/(\varepsilon-1)$}
  Maximizando el valor \rj{$X$} respecto de la barrera \rj{$C^*$}:
  \begin{ec}
    \frac{dX}{dC^*} = 0 \;\;\Longrightarrow\;\;
    \az{(1-\varepsilon)\,C^* + \varepsilon = 0}
    \;\;\Longrightarrow\;\;
    \rj{C^* = \frac{\varepsilon}{\varepsilon-1}}
  \end{ec}
  Como \rj{$\varepsilon>1$}, resulta \rj{$C^*>1$}: el umbral Ăłptimo
  \resultado{siempre supera} el punto donde el VAN se anula.
\end{frame}

% E8 â€” aclaraciones
\begin{frame}{Aclaraciones sobre el modelo general}
  \begin{itemize}
    \item Si \az{$F$} es no estocĂˇstico, \rj{$X$} coincide con la fĂłrmula de
          \concepto{Samuelson (1970)} para una call americana de vida
          infinita sobre una acciĂłn que paga dividendos.
    \item Si $T$ es finito, o si los parĂˇmetros dependen de \rj{$V$}, \az{$F$}
          y $t$, \resultado{no hay soluciĂłn analĂ­tica}: se recurre a
          aproximaciones numĂ©ricas (Ingersoll, 1977).
  \end{itemize}
\end{frame}

% E9 â€” estĂˇticas comparativas analĂ­ticas (OJO: se solapa con AgustĂ­n/Tablas I-II)
\begin{frame}{EstĂˇticas comparativas}
  \begin{itemize}
    \item \rj{$X$} y \rj{$C^*$} son \resultado{crecientes en la varianza} de
          \rj{$V$}$/$\az{$F$}: mĂˇs varianza amplĂ­a la ganancia mĂˇxima y deja
          igual la pĂ©rdida mĂˇxima.
    \item Solo interviene \az{$\sigma^2 = \sigma_v^2+\sigma_f^2-2\rho_{vf}

          \az{$\sigma_f^2$}, o una caĂ­da de \az{$\rho_{vf}$}, eleva el valor
          de la opciĂłn.
    \item \rj{$X$} es creciente en \az{$\alpha_v$} y decreciente en
          \az{$\alpha_f$} y \az{$\mu$}.
  \end{itemize}
  \begin{cajatexto}

  \end{cajatexto}
\end{frame}
